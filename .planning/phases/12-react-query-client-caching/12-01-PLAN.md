From src/app/(app)/layout.tsx (поточний):
```typescript
// Server Component — НЕ має "use client"
import { auth } from "@/auth";
import { AppShell } from "@/components/AppShell";
import { redirect } from "next/navigation";

export default async function ProtectedAppLayout({ children }) {
  const session = await auth();
  if (!session?.user?.id) redirect("/auth/login");
  return <AppShell user={session.user}>{children}</AppShell>;
}
```

Проблема: layout.tsx є Server Component, тому QueryClientProvider (який вимагає "use client") не можна додати напряму.
